      def set_value(value)
        klass = @klass
        const_id = @const_id

        ret = @hook_handler.handle_cdecl( @klass, @const_id, value )

        if ret then
          klass = ret.klass
          const_id = ret.const_id
          value = ret.value
        end

        klass.const_set(const_id, value)
      end
